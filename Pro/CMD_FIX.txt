class FixedAudioPlayer {
  constructor() {
    this.currentAudio = null;
    this.currentTrackId = null;
    this.audioContext = null;
    this.analyser = null;
    this.setupAudioAnalysis();
  }
  
  setupAudioAnalysis() {
    // Crear contexto de audio para K-Meter
    this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
    this.analyser = this.audioContext.createAnalyser();
    this.analyser.fftSize = 512;
    this.analyser.smoothingTimeConstant = 0.3;
    
    this.dataArray = new Float32Array(this.analyser.fftSize);
  }
  
  play(trackPath, trackId, trackData) {
    // Detener audio anterior
    if (this.currentAudio) {
      this.currentAudio.pause();
    }
    
    // Crear nuevo audio
    this.currentAudio = new Audio(trackPath);
    this.currentTrackId = trackId;
    
    // Conectar al analizador
    const source = this.audioContext.createMediaElementSource(this.currentAudio);
    source.connect(this.analyser);
    this.analyser.connect(this.audioContext.destination);
    
    // Actualizar UI con info del track
    this.updateTrackInfo(trackData);
    
    // Reproducir
    this.currentAudio.play().then(() => {
      this.startAnalysis();
      this.updateControls();
    });
    
    // Eventos
    this.currentAudio.addEventListener('timeupdate', () => {
      this.updateProgress();
    });
    
    this.currentAudio.addEventListener('loadedmetadata', () => {
      this.updateDuration();
    });
  }
  
  updateTrackInfo(trackData) {
    // Actualizar título
    const titleElement = document.getElementById('current-title');
    if (titleElement) {
      titleElement.textContent = trackData.title || 'Unknown Title';
    }
    
    // Actualizar artista
    const artistElement = document.getElementById('current-artist');
    if (artistElement) {
      artistElement.textContent = trackData.artist || 'Unknown Artist';
    }
    
    // Actualizar artwork
    const artworkElement = document.getElementById('current-artwork');
    if (artworkElement) {
      artworkElement.src = trackData.artwork || 'assets/default-album.png';
      artworkElement.onerror = () => {
        artworkElement.src = 'assets/default-album.png';
      };
    }
  }
  
  startAnalysis() {
    const animate = () => {
      if (!this.currentAudio || this.currentAudio.paused) return;
      
      // Obtener datos de audio
      this.analyser.getFloatTimeDomainData(this.dataArray);
      
      // Calcular levels para L y R (simulado)
      let peakL = 0, peakR = 0;
      const halfLength = this.dataArray.length / 2;
      
      // Canal izquierdo
      for (let i = 0; i < halfLength; i++) {
        const abs = Math.abs(this.dataArray[i]);
        if (abs > peakL) peakL = abs;
      }
      
      // Canal derecho
      for (let i = halfLength; i < this.dataArray.length; i++) {
        const abs = Math.abs(this.dataArray[i]);
        if (abs > peakR) peakR = abs;
      }
      
      // Convertir a dB y actualizar meters
      const dbL = this.amplitudeToDb(peakL);
      const dbR = this.amplitudeToDb(peakR);
      
      this.updateMeter('left', dbL);
      this.updateMeter('right', dbR);
      
      requestAnimationFrame(animate);
    };
    
    animate();
  }
  
  amplitudeToDb(amplitude) {
    if (amplitude === 0) return -60;
    const db = 20 * Math.log10(amplitude);
    return Math.max(-60, Math.min(0, db));
  }
  
  updateMeter(channel, dbValue) {
    const fillElement = document.getElementById(`level-${channel}`);
    const valueElement = document.getElementById(`value-${channel}`);
    const peakElement = document.getElementById(`peak-${channel}`);
    
    if (fillElement) {
      const percentage = ((dbValue + 60) / 60) * 100;
      fillElement.style.width = `${Math.max(0, percentage)}%`;
    }
    
    if (valueElement) {
      valueElement.textContent = dbValue <= -60 ? '-∞' : dbValue.toFixed(1);
      
      // Color según nivel
      if (dbValue > -6) {
        valueElement.style.color = '#ff0000';
      } else if (dbValue > -20) {
        valueElement.style.color = '#ffff00';
      } else {
        valueElement.style.color = '#00ff00';
      }
    }
    
    if (peakElement) {
      const peakPercentage = ((dbValue + 60) / 60) * 100;
      peakElement.style.left = `${Math.max(0, peakPercentage)}%`;
    }
  }
  
  updateProgress() {
    if (!this.currentAudio) return;
    
    const current = this.currentAudio.currentTime;
    const total = this.currentAudio.duration;
    const percentage = (current / total) * 100;
    
    // Actualizar barra
    const progressBar = document.getElementById('progress-bar');
    const progressHandle = document.getElementById('progress-handle');
    if (progressBar) {
      progressBar.style.width = `${percentage}%`;
    }
    if (progressHandle) {
      progressHandle.style.left = `${percentage}%`;
    }
    
    // Actualizar tiempo
    const currentTimeElement = document.getElementById('current-time');
    if (currentTimeElement) {
      currentTimeElement.textContent = this.formatTime(current);
    }
  }
  
  updateDuration() {
    const totalTimeElement = document.getElementById('total-time');
    if (totalTimeElement && this.currentAudio) {
      totalTimeElement.textContent = this.formatTime(this.currentAudio.duration);
    }
  }
  
  formatTime(seconds) {
    if (isNaN(seconds)) return '0:00';
    const mins = Math.floor(seconds / 60);
    const secs = Math.floor(seconds % 60);
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  }
  
  updateControls() {
    const playBtn = document.getElementById('btn-play-pause');
    const playIcon = playBtn.querySelector('.play-icon');
    const pauseIcon = playBtn.querySelector('.pause-icon');
    
    if (this.currentAudio && !this.currentAudio.paused) {
      playIcon.style.display = 'none';
      pauseIcon.style.display = 'block';
    } else {
      playIcon.style.display = 'block';
      pauseIcon.style.display = 'none';
    }
  }
}
